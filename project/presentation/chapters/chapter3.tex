\begin{frame}{Infrastructure as Code: Two Automated Steps}
\begin{itemize}
    \item \textbf{Provisioning -- OpenTofu:} declaratively creates the Azure VM (compute, disk, networking).
    \item \textbf{Configuration -- Ansible:} installs Docker, MicroK8s and all dependencies on the VM.
    \item \textcolor{HUAblue}{\textbf{The result:}} the entire environment is reproducible from code -- no manual setup, no "works on my machine".
\end{itemize}
\end{frame}

\begin{frame}{Managing Secrets Securely}
\begin{itemize}
    \item \textbf{No secrets in Git:} credentials (MongoDB, MinIO, RabbitMQ) are never committed to the repository.
    \item \textbf{Externalized configuration:} secrets are stored as Kubernetes Secrets and injected into pods at runtime.
    \item \textcolor{HUAblue}{\textbf{Future work:}} centralized secrets with HashiCorp Vault and user authentication via Keycloak (SSO).
\end{itemize}
\end{frame}
