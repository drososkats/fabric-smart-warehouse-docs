\section{Conclusions \& Future Work}

\begin{frame}{Conclusions \& Future Work}
    \vspace{0.2cm}
    
    \begin{columns}[T]
        % --- Left Column: Conclusions ---
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[colback=white, colframe=gray!30, arc=2pt, boxrule=0.5pt]
                \textbf{\textcolor{blue!70!black}{Conclusions}}\\[0.2cm]
                \scriptsize
                \begin{itemize}
                    \setlength{\itemsep}{6pt}
                    \item \textbf{Cloud-native transformation:} Automated, observable, GitOps-driven platform on Azure.
                    \item \textbf{Full GitOps automation:} \texttt{git push} flows seamlessly to the cluster via Jenkins \& ArgoCD.
                    \item \textbf{Resilience by design:} Claim Check, Retry, Idempotency and Stateless patterns.
                    \item \textcolor{HUAblue}{\textbf{Key Takeaway:}} Complexity lives in the \textit{interactions} between technologies, not in isolation.
                \end{itemize}
            \end{tcolorbox}
        \end{column}
        
        % --- Right Column: Future Work ---
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[colback=white, colframe=gray!30, arc=2pt, boxrule=0.5pt]
                \textbf{\textcolor{gray!80!black}{Known Limitations \& Future Work}}\\[0.2cm]
                \scriptsize
                \begin{itemize}
                    \setlength{\itemsep}{6pt}
                    \item \textbf{Circuit Breaker:} Not yet implemented (a natural extension to the Retry mechanism).
                    \item \textbf{RabbitMQ secrets:} Connection relies on env variables due to a known Docker image limitation.
                    \item \textbf{GET pagination:} \texttt{/api/products} returns all records (degrades performance under high load).
                    \item \textcolor{HUAblue}{\textbf{Pod-level FinOps:}} Requires additional tooling (e.g., \textit{OpenCost}) beyond \textit{Infracost}.
                \end{itemize}
            \end{tcolorbox}
        \end{column}
    \end{columns}
 \end{frame}
