\section{Conclusions \& Future Work}

\begin{frame}{Conclusions \& Future Work}
    \begin{itemize}
            \item \textbf{Automation Success}: Fully automated the deployment of 7+ services using \texttt{Bash} scripts and \texttt{Kubernetes YAML} files.
            \item \textbf{Hybrid Efficiency}: Successfully connected local \texttt{MicroK8s} services with the \texttt{MongoDB Atlas} public cloud.
            \item \textbf{Identity Management (Future)}: Integrating \texttt{Keycloak} to provide secure, centralized authentication (\texttt{SSO}) for all services.
            \item \textbf{Data Persistence (Future)}: Using Persistent Volumes (\texttt{PV}) to ensure data is safe if a \texttt{Pod} restarts.
            \item \textbf{Traffic Control (Future)}: Implementing an Ingress Controller for professional \texttt{URL} management and improved security.
        \end{itemize}
\end{frame}
