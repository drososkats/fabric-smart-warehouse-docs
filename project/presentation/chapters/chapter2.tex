% ===== Topic A: Evolution =====
\begin{frame}{The Evolution: From Local Prototype to Cloud-Native}
\begin{itemize}
    \item \textbf{Before:} a single machine, ephemeral data, manual deployment, no resilience.
    \item \textbf{After:} the same application core -- now durable, automated and observable.
    \item \textcolor{HUAblue}{\textbf{The core idea:}} we do not rebuild the application; we add cloud-native layers around a solid base.
\end{itemize}
\end{frame}

% ===== Topic B: System Architecture =====
\begin{frame}{System Architecture: A Hybrid Cloud Approach}
\begin{itemize}
    \item \textbf{Application \& logic (private cloud):} Backend, Frontend and services run inside MicroK8s.
    \item \textbf{Data persistence (public cloud):} MongoDB Atlas (DBaaS) stays outside the cluster for durability.
    \item \textbf{Orchestrated services:} Frontend, Backend, RabbitMQ, MinIO, Node-RED and ThingsBoard, each as a pod.
    \item \textcolor{HUAblue}{\textbf{Automation:}} declarative YAML manifests replace manual configuration.
\end{itemize}
\end{frame}

% ===== Topic C: Data Flow & Persistence =====
\begin{frame}{The Add-Product Flow \& Making It Durable}
\begin{itemize}
    \item \textbf{Event-driven flow:} image to MinIO, metadata to MongoDB, event to RabbitMQ, then Node-RED to ThingsBoard.
    \item \textbf{The persistence fix:} Persistent Volumes (PV/PVC) so data survives pod restarts.
    \item \textcolor{HUAblue}{\textbf{Why it matters:}} this solves the biggest limitation of the original prototype.
\end{itemize}
\end{frame}
