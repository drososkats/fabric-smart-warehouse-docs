\section{Problem Definition \& Goal}

{
    \metroset{titleformat frame=smallcaps}
\begin{frame}[fragile]{Defining the Challenge: From Monolith to Cloud-Native}

\begin{itemize}
    \item \textbf{The Challenge:}  Transitioning a legacy monolithic Inventory ERP to a modern, scalable environment.
    \item \textbf{Current Limitations:} 
    \begin{itemize}
        \item \textbf{Scalability Issues:} Hard to scale individual components without affecting the whole system.
        \item \textbf{Single Point of Failure:} Tightly coupled services increase the risk of complete system downtime.
        \item \textbf{Deployment Complexity:} Environment dependencies make migration to production error-prone.
    \end{itemize}
    \item \textcolor{HUAblue}{\textbf{The Goal:}} To design a distributed, Cloud-Native Smart Warehouse system using Microservices and IoT integration.
\end{itemize}

\end{frame}

}
