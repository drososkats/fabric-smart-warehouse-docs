{
    \metroset{titleformat frame=smallcaps}
\begin{frame}{IoT Integration}

\begin{itemize}
            \item \textbf{The Trigger}: Adding a new product in the \texttt{ERP} app starts a real-time data sequence.
            \item \textbf{Media Storage}: Product images are automatically uploaded to the \texttt{MinIO} bucket.
            \item \textbf{Event Messaging}: \texttt{RabbitMQ} alerts the system that a new entry is ready for processing.
            \item \textbf{Logic Engine}: \texttt{Node-RED} consumes the message and simulates warehouse conditions (Temp/Humidity) using a custom script.
            \item \textbf{Live Dashboard}: All data is pushed to \texttt{ThingsBoard} for real-time monitoring and history tracking.
        \end{itemize}

\end{frame}
