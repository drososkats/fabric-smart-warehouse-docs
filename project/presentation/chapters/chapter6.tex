\section{Serverless Computation}
{
    \metroset{titleformat frame=smallcaps}

% --- Slide 1 (words) ---
\begin{frame}{Serverless: Computation Without Servers}
\begin{itemize}
    \item \textbf{What it is:} code that runs only when an event happens -- no always-on server, resources are used only during execution.
    \item \textbf{Where we use it:} a function triggered by a MinIO "new object" event.
    \item \textcolor{HUAblue}{\textbf{Platform:}} Knative on MicroK8s (lightweight) for event-driven functions.
\end{itemize}
\end{frame}

% --- Slide 2 (words) ---
\begin{frame}{The Trigger in Action}
\begin{itemize}
    \item \textbf{Event:} when a product image is uploaded to MinIO, an event notification fires automatically.
    \item \textbf{Function:} it reads the new image, extracts metadata (size, dimensions) and generates a thumbnail.
    \item \textbf{Result:} the metadata and thumbnail are written back to MinIO/MongoDB -- with no manual step and no extra always-on service.
    \item \textcolor{HUAblue}{\textbf{Benefit:}} heavy, occasional work is offloaded from the always-on backend.
\end{itemize}
\end{frame}

}
