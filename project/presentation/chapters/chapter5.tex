% --- Slide 5: Retry Pattern in Code ---
\begin{frame}[fragile]{Retry Pattern: Implementation}

    \begin{tcolorbox}[colback=blue!3, colframe=blue!40, arc=2pt, boxrule=0.5pt, title=\footnotesize\textbf{Generic Retry Utility (\texttt{retry.js})}]
\ttfamily\scriptsize
\begin{verbatim}
async function retryWithDelay(fn, { retries = 3, delayMs = 2000,
                                     label = "operation" } = {}) {
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      return await fn();
    } catch (err) {
      if (attempt < retries) {
        await new Promise((resolve) => setTimeout(resolve, delayMs));
      } else {
        throw err;
      }
    }
  }
}
\end{verbatim}
    \end{tcolorbox}

    \vspace{0.2cm}

    \begin{tcolorbox}[colback=green!3, colframe=green!40!black, arc=2pt, boxrule=0.5pt, title=\footnotesize\textbf{Applied to: MongoDB Connection (Startup)}]
\ttfamily\scriptsize
\begin{verbatim}
retryWithDelay(
  () => mongoose.connect(MONGO_URI)
          .then(() => console.log("MongoDB connected")),
  { retries: 3, delayMs: 2000, label: "MongoDB connection" }
);
\end{verbatim}
    \end{tcolorbox}

\end{frame}
